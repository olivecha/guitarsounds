 358    def on_ok_button_clicked_2(self, b):
 359        """
 360        Method to make the "Ok" button interact with the
 361        analysis method choice.
 362
 363        __ when interface.state = 'method choice' __
 364        - The "Ok" and "Go" buttons appears after the loading bar is done
 365        - The dropdown corresponds to the methods associated to
 366        the analysis
 367        :param: b the ipywidget button object for which this method is executed when it is clicked
 368        """
 369        if b is not None:
 370            pass
 371        # Clear the Output
 372        clear_output(wait=True)
 373        output = widgets.Output(layout=self.out_layout)
 374
 375        # Save the dropdown value
 376        drop_down_value = self.current_drop_down.value
 377        
 378        # enable the info button when coming back from display
 379        if self.state != 'display':
 380            self.info_button.disabled = False
 381            self.toggle_normalize_button.disabled = False
 382
 383        # Deactivate the info button if it was activated
 384        if self.info_button.button_style == 'info':
 385            self.info_button.button_style = ''
 386
 387        if self.state == 'method choice':  # State when the user is choosing the analysis method
 388
 389            # If we only analyse a single sound
 390            if self.analysis == 'Single':
 391
 392                # Special case when the method is the frequency bin plot
 393                if drop_down_value == Sound.plot_freq_bins:
 394                    self.analysis_tuple = [drop_down_value]  # Store the method
 395                    # Change the dropdown to frequency bin choice
 396                    self.current_drop_down = self.bin_drop_down
 397                    self.state = 'method choice 2'  # a second choice is needed
 398                    self.display = 'plot'
 399
 400                # Case for the methods without plotting
 401                elif drop_down_value in [Sound.peak_damping, Sound.listen_freq_bins, Signal.listen]:
 402                    self.analysis_tuple = [drop_down_value]  # store the method
 403                    self.state = 'display'  # ready to display
 404                    self.display = 'print'
 405
 406                # Signal.plot.method() methods
 407                elif drop_down_value in [*self.plot_methods, Sound.bin_hist]:
 408                    # store method and arg in a list
 409                    self.analysis_tuple = [drop_down_value]
 410                    self.state = 'display'  # ready to display
 411                    self.display = 'plot'
 412
 413                # Error when no method is chosen
 414                elif drop_down_value == 1:
 415                    error = generate_error_widget('No analysis method was chosen')
 416                    with output:
 417                        display(error)
 418
 419            # Case when two sounds or multiple sounds are being analysed
 420            elif self.analysis in ['Dual', 'Multiple']:
 421
 422                # Special case for the frequency bin plot
 423                if drop_down_value in self.DM_bin_choice_methods:
 424                    self.analysis_tuple = [drop_down_value]  # Store the method
 425                    # Update the dropdown to frequency bin choice
 426                    self.current_drop_down = self.bin_drop_down
 427                    self.state = 'method choice 2'  # a second choice is needed
 428                    self.display = 'plot'
 429
 430                # Case for plot methods
 431                elif (drop_down_value == SoundPack.plot) or (drop_down_value == SoundPack.compare_plot):
 432                    self.analysis_tuple = [drop_down_value]  # Store the method
 433                    # Update the dropdown to the plot dropdown
 434                    self.current_drop_down = self.plot_drop_down
 435                    self.state = 'method choice 2'  # a second choice is needed
 436                    self.display = 'plot'
 437
 438                # Error when no method is chosen
 439                elif drop_down_value == 1:
 440                    error = generate_error_widget('No analysis method was chosen')
 441                    with output:
 442                        display(error)
 443
 444                # Case for methods with no arguments
 445                else:
 446                    if drop_down_value == SoundPack.fundamentals:
 447                        self.display = 'print'
 448                    else:
 449                        self.display = 'plot'
 450                    self.analysis_tuple = [drop_down_value]  # store the method
 451                    self.state = 'display'
 452
 453        # Case when the method is chosen and an argument needs to be added 'method choice 2'
 454        elif self.state == 'method choice 2':
 455
 456            # add the arg part to the analysis tuple
 457            self.analysis_tuple.append(self.current_drop_down.value)
 458            self.state = 'display'
 459
 460        # if we are coming back from the display the state is redefined, and we restart
 461        elif self.state == 'analysis displayed':
 462            self.state = 'method choice'
 463
 464        # If the button is pressed and the method is defined, the go button is enabled
 465        if self.state == 'display':
 466            self.go_button.disabled = False
 467            self.ok_button.disabled = True
 468            self.info_button.disabled = True
 469            self.toggle_normalize_button.disabled = True
 470
 471        # Actualize the button box and display
 472        children = [self.ok_button, self.go_button, self.toggle_normalize_button, self.info_button]
 473        self.button_box = widgets.Box(children=children, layout=self.box_layout)
 474
 475        # Put the updated dropdown in the output
 476        with output:
 477            display(self.current_drop_down)
 478
 479        display(self.button_box, output)

 579    def on_go_button_clicked(self, b):
 580        """
 581        Go button to display the analysis when all choices are made
 582
 583        What happens :
 584        ___________________________________
 585        1. The output is cleared
 586        2. An output widget to store the output is instanced
 587        3. The method in `self.analysis_tuple` is called
 588        4. The display is added to the output
 589        5. The 'Ok' button is enabled and the 'Go' button is disabled
 590        6. The dropdown is set back to its default value
 591        7. The buttons and output are displayed
 592        :param b: the ipywidget button object corresponding to the go button
 593        """
 594        if b is not None:
 595            pass
 596        # Always clear the output
 597        clear_output(wait=True)
 598        output = widgets.Output(layout=self.out_layout)  # Create a output
 599
 600        # Change the GUI state
 601        self.state = 'analysis displayed'
 602
 603        # Set the matplotlib display method
 604        get_ipython().run_line_magic('matplotlib', 'inline')
 605
 606        # Case for a single sound
 607        if self.analysis == 'Single':
 608
 609            # Case for Sound.plot_freq_bins method
 610            if self.analysis_tuple[0] == Sound.plot_freq_bins:
 611                # change interface
 612                get_ipython().run_line_magic('matplotlib', 'notebook')
 613                # create a figure
 614                plt.figure(figsize=(8, 6))
 615                # Call the method
 616                self.analysis_tuple[0](self.sounds, bins=[self.analysis_tuple[1]])
 617
 618                # Define the title according to the chosen bin
 619                if self.analysis_tuple[1] == 'all':
 620                    plt.title('Frequency bin plot for ' + self.sounds.name)
 621                else:
 622                    plt.title(self.analysis_tuple[1] + ' bin plot for ' + self.sounds.name)
 623
 624                    plt.show()
 625
 626            # Case for the Sound.peak_damping method (print only)
 627            elif self.analysis_tuple[0] in [Sound.peak_damping, Sound.listen_freq_bins]:
 628                with output:
 629                    self.analysis_tuple[0](self.sounds)  # add print to output
 630
 631            # Case for the Signal.plot method
 632            elif self.analysis_tuple[0] in self.plot_methods:
 633                # change plot interface
 634                get_ipython().run_line_magic('matplotlib', 'notebook')
 635                # create a figure
 636                plt.figure(figsize=(8, 6))
 637                # Add the fill argument if there is just one plot
 638                kwargs = {}
 639                # Call the method according to normalization
 640                if not self.normalize:
 641                    self.analysis_tuple[0](self.sounds.signal.plot, **kwargs)
 642                elif self.normalize:
 643                    self.analysis_tuple[0](self.sounds.signal.normalize().plot, **kwargs)

 580    def on_go_button_clicked(self, b):
 581        """
 582        Go button to display the analysis when all choices are made
 583
 584        What happens :
 585        ___________________________________
 586        1. The output is cleared

 588        3. The method in `self.analysis_tuple` is called
 589        4. The display is added to the output
 590        5. The 'Ok' button is enabled and the 'Go' button is disabled
 591        6. The dropdown is set back to its default value
 592        7. The buttons and output are displayed

 594        """

 596            pass

 599        output = widgets.Output(layout=self.out_layout)  # Create a output
 600
 601        # Change the GUI state
 602        self.state = 'analysis displayed'
 603
 604        # Set the matplotlib display method
 605        get_ipython().run_line_magic('matplotlib', 'inline')
 606
 607        # Case for a single sound
 608        if self.analysis == 'Single':
 609
 610            # Case for Sound.plot_freq_bins method
 611            if self.analysis_tuple[0] == Sound.plot_freq_bins:
 612                # change interface

 621                    plt.title('Frequency bin plot for ' + self.sounds.name)

 623                    plt.title(self.analysis_tuple[1] + ' bin plot for ' + self.sounds.name)

 625                    plt.show()
 626
 627            # Case for the Sound.peak_damping method (print only)
 628            elif self.analysis_tuple[0] in [Sound.peak_damping, Sound.listen_freq_bins]:
 629                with output:
 630                    self.analysis_tuple[0](self.sounds)  # add print to output

 632            # Case for the Signal.plot method

 634                # change plot interface
 635                get_ipython().run_line_magic('matplotlib', 'notebook')
 636                # create a figure
 637                plt.figure(figsize=(8, 6))
 638                # Add the fill argument if there is just one plot
 639                kwargs = {}

 642                    self.analysis_tuple[0](self.sounds.signal.plot, **kwargs)

 644                    self.analysis_tuple[0](self.sounds.signal.normalize().plot, **kwargs)
